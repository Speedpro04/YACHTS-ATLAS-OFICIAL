Distribuição
A maioria está no Sudeste e Sul, com destaque para redes como a BR Marinas, que opera 10 unidades só no litoral do Rio de Janeiro. Outras marinas notáveis incluem as de Santa Catarina (como Itajaí e Tedesco) e São Paulo (como Iate Clube de Santos).

import stripe
import os
from fastapi import FastAPI, Request, HTTPException, Header
from pydantic import BaseModel
from dotenv import load_dotenv

load_dotenv()
stripe.api_key = os.getenv("STRIPE_SECRET_KEY")
app = FastAPI(title="Yachts Atlas Subs")

PRICE_ID = os.getenv("STRIPE_PRICE_ID")  # price_... do dashboard
WEBHOOK_SECRET = os.getenv("STRIPE_WEBHOOK_SECRET")

class CheckoutData(BaseModel):
    email: str
    marina_id: str  # Tenant Supabase

@app.post("/create-checkout/")
async def create_checkout(data: CheckoutData):
    session = stripe.checkout.Session.create(
        payment_method_types=["card"],
        customer_email=data.email,
        line_items=[{"price": PRICE_ID, "quantity": 1}],
        mode="subscription",
        success_url="https://yachtsatlas.com/success?session_id={CHECKOUT_SESSION_ID}",
        cancel_url="https://yachtsatlas.com/cancel",
        metadata={"marina_id": data.marina_id}
    )
    return {"url": session.url}

@app.post("/webhook/")
async def webhook(request: Request, stripe_signature: str = Header(None)):
    payload = await request.body()
    event = stripe.Webhook.construct_event(payload, stripe_signature, WEBHOOK_SECRET)
    
    if event["type"] == "checkout.session.completed":
        session = event["data"]["object"]
        marina_id = session.metadata["marina_id"]
        # Ativa no Supabase (adicione sua lógica)
        print(f"Ativou marina {marina_id}!")
    
    return {"status": "ok"}
